class Solution {
    public int repeatedNTimes(int[] nums) {
        int m = 0;
        HashMap<Integer,Integer> h = new HashMap<>();
        for(int i : nums){
            h.put(i, h.getOrDefault(i, 0) + 1);
            if(h.get(i) == nums.length/2){
                return i;
            }
        }
        return -1;
    }
}

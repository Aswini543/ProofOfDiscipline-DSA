class Solution {
    public int sumFourDivisors(int[] nums) {
        int s = 0;
        for(int i : nums){
            s += solve(i);
        }
        return s;
    }
    public int solve(int num){
        int c = 0,sum=0;
        for(int i = 1;i<=num;i++){
            if(c>4) return 0;
            else{
                if(num%i == 0){
                    sum += i;
                    c++;
                }
            }
        }
        return (c==4) ? sum : 0;
    } 
}
